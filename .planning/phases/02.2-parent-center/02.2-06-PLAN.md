PIN storage contract (localStorage):
- Key: `'parent_pin_hash'`
- Value: hex string — SHA-256 hash of the 4-digit PIN
- Use Web Crypto API: `crypto.subtle.digest('SHA-256', encoder.encode(pin))` → convert to hex
- No external library needed
```typescript
async function hashPin(pin: string): Promise<string> {
  const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(pin))
  return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2, '0')).join('')
}
```

**PIN gate logic:**

```typescript
const [pinUnlocked, setPinUnlocked] = useState(false)
const [pinInput, setPinInput] = useState('')
const [pinError, setPinError] = useState('')
const [isSettingPin, setIsSettingPin] = useState(false)

useEffect(() => {
  const stored = localStorage.getItem('parent_pin_hash')
  if (!stored) setIsSettingPin(true)  // first time — create PIN flow
}, [])

async function handlePinSubmit() {
  const hash = await hashPin(pinInput)
  const stored = localStorage.getItem('parent_pin_hash')
  if (isSettingPin) {
    // Creating new PIN
    localStorage.setItem('parent_pin_hash', hash)
    setPinUnlocked(true)
  } else {
    if (hash === stored) {
      setPinUnlocked(true)
    } else {
      setPinError('Неверный PIN')
      setPinInput('')
    }
  }
}
```
