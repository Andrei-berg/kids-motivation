**Data loading** (triggered when `selectedChildId` changes):
```typescript
// Build array of 8 week starts, newest first
const weeks: string[] = []
for (let i = 7; i >= 0; i--) {
  const d = new Date()
  d.setDate(d.getDate() - i * 7)
  weeks.push(getWeekRange(d).start)
}

// Fetch all 8 weeks in parallel
const scores = await Promise.all(
  weeks.map(w => getWeekScore(selectedChildId, w))
)
```
