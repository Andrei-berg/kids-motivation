```typescript
export async function postSystemMessage(params: {
  familyId: string
  content: string
}): Promise<void> {
  const { error } = await supabase
    .from('chat_messages')
    .insert({
      family_id: params.familyId,
      sender_id: 'system',
      sender_name: 'FamilyCoins',
      sender_role: 'parent',
      message_type: 'system',
      content: params.content,
    })
  if (error) throw error
}
```
