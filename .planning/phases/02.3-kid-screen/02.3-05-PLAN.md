Compute `canAfford_forSelected()` and `shortfall_forSelected()` as small inline functions using `wallet?.coins` and `selectedReward.price_coins`.

**handlePurchase**:
```ts
async function handlePurchase() {
  if (!activeMemberId || !selectedReward) return
  setPurchasing(true)
  try {
    await purchaseReward(activeMemberId, selectedReward.id)
    setSelectedReward(null)
    showToast('Запрос отправлен! 📨 Ожидай одобрения родителя')
    // Reload data
    const [walletData, purchasesData] = await Promise.all([
      getWallet(activeMemberId),
      getPurchases(activeMemberId),
    ])
    setWallet(walletData)
    setPurchases(purchasesData)
  } catch (err: any) {
    showToast(`Ошибка: ${err.message}`)
  } finally {
    setPurchasing(false)
  }
}
```

**Toast** (fixed top notification):
