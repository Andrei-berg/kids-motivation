    Add a `useCountUp` hook at the top of Dashboard.tsx (before any component definitions). Use the same rAF pattern as the existing `AnimatedNum` in `components/kid/design/atoms.tsx` — do NOT use framer-motion for count-up (simpler, no dependency on animation system):

    ```tsx
    import { useState, useEffect, useRef } from 'react'

    function useCountUp(target: number, duration = 800) {
      const [n, setN] = useState(0)
      const prev = useRef(0)
      useEffect(() => {
        const from = 0  // always count from 0 on first mount
        const to = target
        if (from === to) { setN(to); return }
        const t0 = performance.now()
        let raf: number
        const tick = () => {
          const t = Math.min(1, (performance.now() - t0) / duration)
          const e = 1 - Math.pow(1 - t, 3)  // cubic ease-out
          setN(Math.round(from + (to - from) * e))
          if (t < 1) raf = requestAnimationFrame(tick)
          else prev.current = to
        }
        raf = requestAnimationFrame(tick)
        return () => cancelAnimationFrame(raf)
      }, [target, duration])
      return n
    }
    ```

    ```tsx
    function ChildCard({ child, onAction }: ...) {

    }
    ```
