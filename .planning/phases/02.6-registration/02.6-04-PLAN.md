**Step 2: Create app/api/set-child-pin/route.ts**

This API route handles PIN creation server-side (service-role key must stay server-only):

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { createClient as createBrowserClient } from '@/lib/supabase/client'
import { createClient } from '@supabase/supabase-js'

export async function POST(req: NextRequest) {
  const { memberId, childId, pin } = await req.json()

  if (!memberId || !childId || !pin || !/^\d{4}$/.test(pin)) {
    return NextResponse.json({ error: 'Invalid input' }, { status: 400 })
  }

  const adminClient = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!
  )

  const syntheticEmail = `child_${childId}@internal.familycoins.app`

  // Create auth user (idempotent: if already exists, update password)
  const { data: existingUsers } = await adminClient.auth.admin.listUsers()
  const existing = existingUsers?.users?.find(u => u.email === syntheticEmail)

  if (existing) {
    await adminClient.auth.admin.updateUserById(existing.id, { password: pin })
  } else {
    const { error: createError } = await adminClient.auth.admin.createUser({
      email: syntheticEmail,
      password: pin,
      email_confirm: true,
    })
    if (createError) {
      return NextResponse.json({ error: createError.message }, { status: 500 })
    }
  }

  // Store PIN hash marker in family_members (non-null signals PIN is set)
  // We store a bcrypt-style marker so UI knows PIN exists without storing plaintext.
  // Actual auth is via Supabase auth — child_pin_hash is a UI hint only.
  const supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!
  )
  const { error: updateError } = await supabase
    .from('family_members')
    .update({ child_pin_hash: 'set' })  // marker: 'set' means PIN is configured
    .eq('id', memberId)

  if (updateError) {
    return NextResponse.json({ error: updateError.message }, { status: 500 })
  }

  return NextResponse.json({ ok: true })
}
```

```typescript
/**
 * Set or update a 4-digit PIN for a child (no-email login).
 * Calls /api/set-child-pin which uses service-role key server-side.
 * Safe to call from client components.
 */
export async function setChildPin(memberId: string, childId: string, pin: string): Promise<void> {
  const res = await fetch('/api/set-child-pin', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ memberId, childId, pin }),
  })
  if (!res.ok) {
    const { error } = await res.json().catch(() => ({ error: 'Unknown error' }))
    throw new Error(error || 'Failed to set PIN')
  }
}
```

**lookupFamily() handler (Step 0 next):**

```typescript
async function lookupFamily() {
  setLoading(true)
  setError(null)
  try {
    const result = await lookupFamilyByCode(code.toUpperCase())
    if (!result) {
      setError('Код не найден. Проверь и попробуй снова.')
      return
    }
    const children = await getFamilyChildren(result.familyId)
    setFamilyId_local(result.familyId)
    setProfiles(children)
    setStep(1)
  } catch {
    setError('Ошибка соединения. Попробуй снова.')
  } finally {
    setLoading(false)
  }
}
```

**signInWithPin() handler (Step 2 login):**
```typescript
async function signInWithPin() {
  if (!selectedProfile || pin.length !== 4) return
  setLoading(true)
  setError(null)
  try {
    // child_id is stored in selectedProfile — but ChildProfile only has memberId/displayName.
    // Query family_members to get child_id for this memberId:
    const supabase = createClient()
    const { data: memberRow } = await supabase
      .from('family_members')
      .select('child_id')
      .eq('id', selectedProfile.memberId)
      .maybeSingle()

    if (!memberRow?.child_id) {
      setError('Профиль не найден. Обратись к родителю.')
      return
    }

    const syntheticEmail = `child_${memberRow.child_id}@internal.familycoins.app`
    const { error: authError } = await supabase.auth.signInWithPassword({
      email: syntheticEmail,
      password: pin,
    })

    if (authError) {
      setError('Неверный PIN. Попробуй ещё раз.')
      return
    }

    setFamilyId(familyId)
    setActiveMemberId(memberRow.child_id)
    router.push('/kid/day')
  } catch {
    setError('Ошибка входа. Попробуй снова.')
  } finally {
    setLoading(false)
  }
}
```
