**`handleFinish(customCoinRules?: typeof coinRules)` function** (called from Step 3):
```typescript
async function handleFinish(customCoinRules?: typeof coinRules) {
  setSaving(true)
  setError(null)
  try {
    const supabase = createClient()
    const { data: { user } } = await supabase.auth.getUser()
    if (!user) throw new Error('Не авторизован')

    // 1. Create family + parent member (createFamily upserts parent family_members row —
    //    safe even if auth callback pre-created it; no duplicate parent row risk)
    // Pass parentName to createFamily to set display_name on the parent family_members row (REQ-ONB-003).
    // /register does not capture display_name — this wizard step is the sole collection point.
    // Before calling createFamily, update its signature in lib/onboarding-api.ts to accept
    //  and pass it through to the
    // family_members INSERT/UPSERT as display_name. If updateFamily signature is inconvenient,
    // alternatively call supabase.from('family_members').update({ display_name: parentName.trim() })
    // .eq('user_id', user.id).eq('family_id', familyId) immediately after createFamily returns.
    const { familyId, inviteCode: code } = await createFamily(user.id, { name: familyName.trim(), parentDisplayName: parentName.trim() })

    // 2. Create each child with wallet (sequential, order matters for FK)
    for (const child of children.filter(c => c.name.trim())) {
      await createChildWithWallet(familyId, { name: child.name.trim() })
    }

    // 3. Insert wallet_settings defaults (pass overrides if user customized)
    await completeOnboarding(familyId, customCoinRules)

    setInviteCode(code)
    setStep(4)  // advance to Done screen
  } catch (err) {
    setError(err instanceof Error ? err.message : 'Произошла ошибка. Попробуйте снова.')
  } finally {
    setSaving(false)
  }
}
```
