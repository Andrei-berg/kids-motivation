```typescript
'use client'
import { createClient } from '@/lib/supabase/client'

export interface AuditEvent {
  id: string
  family_id: string
  child_id: string | null
  action_type: 'shop_approve' | 'shop_reject' | 'coin_adjust' | 'badge_award' | 'settings_change' | 'data_export' | 'account_delete_request'
  description: string
  coins_delta: number | null
  actor_user_id: string | null
  metadata: Record<string, unknown>
  created_at: string
}

export type InsertAuditEventParams = Omit<AuditEvent, 'id' | 'created_at'>

export async function insertAuditEvent(params: InsertAuditEventParams): Promise<void> {
  const supabase = createClient()
  const { error } = await supabase.from('parent_audit_events').insert(params)
  if (error) {
    // Non-blocking: log but don't throw — audit failure must not break the parent action
    console.error('[audit] insertAuditEvent failed', error)
  }
}

export async function getAuditEvents(
  familyId: string,
  opts: { childId?: string; limit?: number; offset?: number } = {}
): Promise<AuditEvent[]> {
  const supabase = createClient()
  let q = supabase
    .from('parent_audit_events')
    .select('*')
    .eq('family_id', familyId)
    .order('created_at', { ascending: false })
    .limit(opts.limit ?? 50)

  if (opts.childId) q = q.eq('child_id', opts.childId)
  if (opts.offset) q = q.range(opts.offset, opts.offset + (opts.limit ?? 50) - 1)

  const { data, error } = await q
  if (error) throw error
  return (data ?? []) as AuditEvent[]
}
```

Key constraint: insertAuditEvent is non-blocking (catches errors, logs, does not throw) so it never breaks the parent action it's called after.
