  <action>
    Create `components/PageTransition.tsx` as a 'use client' component. This must be a client component because AnimatePresence uses React context.

    Implementation:
    ```tsx
    'use client'
    import { AnimatePresence, motion } from 'framer-motion'
    import { usePathname } from 'next/navigation'

    export function PageTransition({ children }: { children: React.ReactNode }) {
      const pathname = usePathname()
      return (
        <AnimatePresence mode="wait" initial={false}>
          <motion.div
            key={pathname}
            initial={{ opacity: 0, scale: 0.98 }}
            animate={{ opacity: 1, scale: 1 }}
            exit={{ opacity: 0, scale: 0.98 }}
            transition={{ duration: 0.15, ease: 'easeOut' }}
            style={{ minHeight: '100%' }}
          >
            {children}
          </motion.div>
        </AnimatePresence>
      )
    }
    ```

    Key decisions:
    - `mode="wait"` — exit completes before next page enters (prevents overlap flash)
    - `initial={false}` — no animation on first mount (avoids fade-in on hard reload)
    - `duration: 0.15` — 150ms matches `--transition-fast` CSS variable
    - `scale: 0.98 → 1` — subtle scale per user decision (not 0 → 1 which would be jarring)
    - Framer Motion natively respects `prefers-reduced-motion` — users with reduced motion see instant transitions automatically, no extra code needed
    - `style={{ minHeight: '100%' }}` — wrapper div must not collapse layout; both parent and kid layouts use `min-h-screen` on their own div, so this is safe

    Then update `app/layout.tsx` to import and wrap `{children}` with `<PageTransition>`:
    ```tsx
    import { PageTransition } from '@/components/PageTransition'
    // ...
    <body>
      <OfflineBanner />
      <PushInit />
      <InstallPrompt />
      <PageTransition>
        {children}
      </PageTransition>
    </body>
    ```

    Do NOT wrap OfflineBanner/PushInit/InstallPrompt inside PageTransition — those are persistent UI elements that must not animate out on navigation.
  </action>
